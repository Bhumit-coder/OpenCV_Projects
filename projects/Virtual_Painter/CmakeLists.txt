cmake_minimum_required(VERSION 3.10)
project(Virtual_Painter)

find_package(OpenCV REQUIRED)

add_executable(Virtual_Painter
    src/main.cpp
    src/HSVCalibrator.cpp
    src/VirtualPainter.cpp
)

target_include_directories(Virtual_Painter PRIVATE src)
target_link_libraries(Virtual_Painter ${OpenCV_LIBS})
